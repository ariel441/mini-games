// משחק ניחוש מספר (משחק בודד)
const randomNumber = Math.floor(Math.random() * 10) + 1;

function guessNumber() {
const userGuess = parseInt(document.getElementById('guessInput').value);
const resultP = document.getElementById('result');

if (isNaN(userGuess)) {
resultP.textContent = 'אנא הכנס מספר תקין.';
return;
}

if (userGuess === randomNumber) {
resultP.textContent = 'כל הכבוד! ניחשת נכון!';
let reward = 5;
currentUser ? currentUser.coins += reward : guestCoins += reward;
updateCoinDisplay();
} else {
resultP.textContent = userGuess < randomNumber ? 'המספר גבוה יותר.' : 'המספר נמוך יותר.';
}
}

// לשוניות בעמוד "משחקונים"
function showGameTab(tab) {
document.getElementById('playSection').style.display = tab === 'play' ? 'block' : 'none';
document.getElementById('shopSection').style.display = tab === 'shop' ? 'block' : 'none';
document.getElementById('gameTab').classList.toggle('active', tab === 'play');
document.getElementById('shopTab').classList.toggle('active', tab === 'shop');
if (tab === 'shop') renderShopItems();
}

// חנות
const shopItems = [
{ id: 1, name: "רמזור קסמים", cost: 10 },
{ id: 2, name: "כובע מנצח", cost: 20 },
{ id: 3, name: "נעליים מהירות", cost: 15 }
];

function renderShopItems() {
const container = document.getElementById('shopContainer');
container.innerHTML = '';
shopItems.forEach(item => {
const itemDiv = document.createElement('div');
itemDiv.className = 'shop-item';
itemDiv.innerHTML = `<strong>${item.name}</strong><br>עלות: ${item.cost} מטבעות
<button onclick="purchaseItem(${item.id})">קנה</button>`;
container.appendChild(itemDiv);
});
}

function purchaseItem(itemId) {
const item = shopItems.find(i => i.id === itemId);
const shopMessage = document.getElementById('shopMessage');
let coins = currentUser ? currentUser.coins : guestCoins;

if (coins >= item.cost) {
currentUser ? currentUser.coins -= item.cost : guestCoins -= item.cost;
updateCoinDisplay();
shopMessage.textContent = `רכשת בהצלחה את ${item.name}!`;
} else {
shopMessage.textContent = 'אין לך מספיק מטבעות לרכישה.';
}
}

// לוגיקת התחברות והרשמה
function showAuthForm(formType) {
document.getElementById('loginForm').style.display = formType === 'login' ? 'block' : 'none';
document.getElementById('registerForm').style.display = formType === 'register' ? 'block' : 'none';
document.getElementById('loginTab').classList.toggle('active', formType === 'login');
document.getElementById('registerTab').classList.toggle('active', formType === 'register');
}
